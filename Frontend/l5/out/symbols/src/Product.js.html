1 /**
  2  * Creates a new Product
  3  * @class
  4  * @param {string} name - Self explanatory
  5  * @param {number} count - how many are available
  6  * @param {Date} purchase_date - by this time the product should have been sold
  7  * @param {string} status - is sold
  8  * @param {number} [price=undefined] - product's price
  9  * 
 10 */
 11 export function Product(name, count, purchaseDate, status, price = undefined) {
 12    this.setName = function(name) {
 13       this.name = name;
 14    }
 15    this.setCount = function (count) {
 16       this.count = parseInt(count);
 17    }
 18    this.setPurchaseDate = function (purchaseDate) {
 19       this.purchaseDate = Date.parse(purchaseDate)
 20    }
 21    this.setStatus = function (status) {
 22       this.status = JSON.parse(status.toLowerCase());
 23    }
 24    this.setPrice = function (price) {
 25       if(this.status == true)
 26          this.price = parseFloat(price);
 27       else
 28          throw "Can't set price to unsold item"
 29    }
 30    this.setters = {
 31       "name": this.setName.bind(this),
 32       "count": this.setCount.bind(this),
 33       "purchaseDate": this.setPurchaseDate.bind(this),
 34       "price": this.setPrice.bind(this),
 35       "status": this.setStatus.bind(this)
 36    }
 37    this.setName(name);
 38    this.setCount(count);
 39    this.setPurchaseDate(purchaseDate);
 40    this.setStatus(status);
 41    if (price != undefined)
 42       this.setPrice(price);
 43    return this;
 44 }
