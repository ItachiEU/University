1 import { v4 } from 'uuid';
  2 
  3 /**
  4  * Creates a new list of Products
  5  * @class
  6 */
  7 export function List() {
  8    this.content = function(){
  9       return this.list;
 10    }
 11 
 12    /**  @type {Array.Product}*/
 13    this.list = [];
 14 
 15    /**
 16     * Adds a product to the list
 17     * @param {Product} product 
 18     * @returns {string}
 19    */
 20    this.addProduct = function (product) {
 21       product.id = v4();
 22       this.list.push(product);
 23       return product.id;
 24    };
 25 
 26    /**
 27     * Removes a product with given ID
 28     * @param {string} id 
 29    */
 30    this.removeProduct = function(id) {
 31       let index = this.list.map(e => e.id).indexOf(id);
 32       if (index == -1)
 33          return;
 34       this.list.splice(index, 1);      
 35    }
 36 
 37    /**
 38     * Edits a product with given id
 39     * @param {string} id 
 40     * @param {string} paramName
 41     * @param {string} newValue
 42    */
 43    this.editProduct = function(id, paramName, newValue) {
 44       let index = this.list.map(e => e.id).indexOf(id);
 45       if (index == -1)
 46          return;
 47       
 48       if (Object.keys(this.list[index].setters).includes(paramName)) {
 49          this.list[index].setters[paramName](newValue);
 50       }
 51       else {
 52          this.list[index][paramName] = newValue;
 53       }
 54    }
 55 
 56    /**
 57     * Swaps two products on given indices
 58     * @param {number} index1 
 59     * @param {number} index2 
 60     */
 61    this.swapProducts = function (index1, index2) {
 62       if (index1 < 0 || index1 >= this.list.length)
 63          return;
 64       if (index2 < 0 || index2 >= this.list.length)
 65          return;
 66       let p = this.list[index1];
 67       this.list[index1] = this.list[index2];
 68       this.list[index2] = p;
 69    }
 70 
 71    
 72    /**
 73     * Filters products that should be sold today
 74     * @returns {Products[]} - products that should be sold on this date
 75     */
 76    this.getProductsForToday = function(){
 77       let date = new Date();
 78       let [day, month, year] = [date.getDate(), date.getMonth(), date.getFullYear()];
 79       return this.list.filter(el => {
 80          let t = new Date(el.purchaseDate);
 81          return t.getFullYear() == year && t.getMonth() == month && t.getDate() == day;
 82       })
 83    }
 84 
 85    /**
 86     * Returns some of all prices of products sold on given date
 87     * @param {Date} date 
 88     * @returns {number}
 89     */
 90    this.getExpenseSummaryForDate = function (date) {
 91       let [day, month, year] = [date.getDate(), date.getMonth(), date.getFullYear()];
 92       let soldProductsFromDate = this.list.filter(el => {
 93          let t = new Date(el.purchaseDate);
 94          return t.getFullYear() == year && t.getMonth() == month && t.getDate() == day;
 95       }).filter(el => el.status);
 96 
 97       let result = -1;
 98       result = soldProductsFromDate.reduce((accumulator, currentValue) => {
 99          if (currentValue.price != undefined)
100             return accumulator + currentValue.price * currentValue.count;
101          else
102             throw 'Undefined price';
103       }, 0);
104       return result;         
105    }
106 
107    /**
108     * Performs the action defined in callback on products with ids 
109     * specified in the listOfIDs
110     * @param {String[]} listOfIDs 
111     * @param {callback} callback - accepts single Product and modifies it in place
112     */
113    this.massEditProducts = function (listOfIDs, callback) {
114       this.list.filter(el => listOfIDs.includes(el.id)).map(callback);
115    }
116 }
